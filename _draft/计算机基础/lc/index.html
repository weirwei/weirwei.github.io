class Solution:
    def longestPalindrome(self, s: str) -> int:
        dic = {}
        for c in s:
            if c in dic:
                dic[c] += 1
            else:
                dic[c] = 1
        res = 0
        for _, v in dic.items():
            res += v - v % 2
        return res if res == len(s) else res + 1

# -*- coding: UTF-8 -*-

"""
10个货架，每个有不同的容量， 现在需要存储x容量物品， 最少需要多少货架， 且需要的货架最好不空。返回需要的货架坐标的index。
货架容量是从小到大排序好的
[5, 4, 3, 3, 2, 1] 10 [0, 2, 4]
[5, 4, 3, 3, 2, 1] 11 [0, 2, 3]
"""
class Solution:

    def slover(self, lst, cap):
        self.enumLst = []
        self.dfs(lst, cap, [], 0)
        self.enumLst = sorted(self.enumLst, key=lambda v: (-v[0], len(v[1])), reverse=False)
        return self.enumLst[0]

    def dfs(self, lst, needCap, idxs, curIdx):
        if curIdx >= len(lst):
            if needCap <= 0:
                self.enumLst.append((needCap, idxs))
            return 
        self.dfs(lst, needCap - lst[curIdx], idxs + [curIdx], curIdx + 1)
        self.dfs(lst, needCap, idxs, curIdx + 1)


if __name__ == '__main__':

    s = Solution()
    print(s.slover(tmp, 10))

class Solution:
    def repeatedSubstringPattern(self, s: str) -> bool:
        sub = ""
        for i in range(len(s) // 2):
            sub += s[i]
            if sub * (len(s) // (i + 1)) == s:
                return True
        return False

# M*N矩阵 每行升序 找第k大的值
class Solution:
    def theKrdMaxValueOnMatrix(self, mat, k):
        if len(mat) == 0:
            return None
        if k <= 0 or len(mat) * len(mat[0]) < k:
            return None
        mp = [len(mat[0]) - 1 for _ in range(len(mat))]  # 每行最大值的索引
        # 去掉最大的前k-1个元素
        for i in range(k - 1):
            curRow = 0
            for row in range(1, len(mat)):
                if mp[row] >= 0 and mat[curRow][mp[curRow]] < mat[row][mp[row]]:
                    curRow = row
                mp[curRow] -= 1
        # 找每行最大的元素作比较
        theKrdRowIdx = 0
        for row in range(1, len(mat)):
            if mp[row] >= 0 and mat[theKrdRowIdx][mp[theKrdRowIdx]] < mat[row][mp[row]]:
                theKrdRowIdx = row
        return mat[theKrdRowIdx][mp[theKrdRowIdx]]

class Solution:
    def backToOrigin(self,n):
        #点的个数为10
        length = 10
        dp = [[0 for i in range(length)] for j in range(n+1)]
        dp[0][0] = 1
        for i in range(1,n+1):
            for j in range(length):
                #dp[i][j]表示从0出发，走i步到j的方案数
                dp[i][j] = dp[i-1][(j-1+length)%length] + dp[i-1][(j+1)%length]
        return dp[n][0]

class Solution:
    def search(self, nums: List[int], target: int) -> int:
        left = 0
        right = len(nums) - 1
        while left <= right:
            mid = left + (right - left) // 2
            if nums[mid] == target:
                return mid
            if nums[0] <= nums[mid]:
                if nums[0] <= target and target < nums[mid]:
                    right = mid - 1
                else:
                    left = mid + 1
            else:
                if nums[mid] < target and target <= nums[len(nums) - 1]:
                    left = mid + 1
                else:
                    right = mid - 1
        return -1

class Solution:
    def spiralOrder(self, matrix: List[List[int]]) -> List[int]:
        res = []
        if len(matrix) == 0: return res
        left = 0
        right = len(matrix[0]) - 1
        up = 0
        down = len(matrix) - 1
        num = len(matrix) * len(matrix[0])
        while num > 0:
            for i in range(left, right + 1):
                res.append(matrix[up][i])
                num -= 1
            for j in range(up + 1, down + 1):
                res.append(matrix[j][right])
                num -= 1
            if up == down:
                break
            for i in range(right - 1, left - 1, -1):
                res.append(matrix[down][i])
                num -= 1
            if left == right:
                break
            for j in range(down - 1, up, -1):
                res.append(matrix[j][left])
                num -= 1
            left += 1
            right -= 1
            up += 1
            down -= 1
        return res

class Solution:
    def reorderList(self, head: ListNode) -> None:
        # 找mid
        head1 = ListNode()
        head2 = ListNode()
        slow = head
        fast = head
        while fast.next is not None and fast.next.next is not None:
            slow = slow.next
            fast = fast.next.next

        # 拆分
        head1.next = head
        head2.next = slow.next
        slow.next = None



        # 翻转
        pre = None
        cur = head2.next
        while cur is not None:
            tmp  = cur.next
            cur.next = pre
            pre = cur
            cur = tmp
        head2.next = pre

        # merge
        res = ListNode()
        p = res
        p1 = head1.next
        p2 = head2.next


        while p1 is not None and p2 is not None:
            p.next = p1
            p1 = p1.next
            p = p.next
            p.next = p2
            p2 = p2.next
            p = p.next


        if p1 is None:
            p.next = p2
        else:
            p.next = p1
        return res.next

class Solution:
    def trap(self, height: List[int]) -> int:
        if len(height) == 0:
            return 0
        maxLeftHigh = height[0]
        maxRightHigh = height[len(height) - 1]
        left = 0
        right = len(height) - 1
        res = 0
        while left < right:
            maxLeftHigh = max(maxLeftHigh, height[left])
            maxRightHigh = max(maxRightHigh, height[right])
            if height[left] < height[right]:
                res += maxLeftHigh - height[left]
                left += 1
            else:
                res += maxRightHigh - height[right]
                right -= 1
        return res

32. 最长有效括号

具体做法是我们始终保持栈底元素为当前已经遍历过的元素中「最后一个没有被匹配的右括号的下标」，这样的做法主要是考虑了边界条件的处理，栈里其他元素维护左括号的下标：

对于遇到的每个 \text{‘(’}‘(’ ，我们将它的下标放入栈中 对于遇到的每个 \text{‘)’}‘)’ ，我们先弹出栈顶元素表示匹配了当前右括号： 如果栈为空，说明当前的右括号为没有被匹配的右括号，我们将其下标放入栈中来更新我们之前提到的「最后一个没有被匹配的右括号的下标」 如果栈不为空，当前右括号的下标减去栈顶元素即为「以该右括号为结尾的最长有效括号的长度」

class Solution:
    def longestValidParentheses(self, s: str) -> int:
        stk = [-1]
        res = 0
        for i in range(len(s)):
            if s[i] == "(":
                stk.append(i)
            else:
                stk.pop()
                if stk == []:
                    stk.append(i)
                else:
                    res = max(res, i - stk[-1])
        return res

class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        if len(nums) == 0:
            return 0
        dp = []
        for i in range(len(nums)):
            dp.append(1)

                if nums[j] < nums[i]:
                    dp[i] = max(dp[i], dp[j] + 1)
        return max(dp)

class Solution:
    def maximalSquare(self, matrix: List[List[str]]) -> int:
        if len(matrix) == 0:
            return 0
        dp = [[0 for _ in range(len(matrix[0]))] for _ in range(len(matrix))]
        maxSide = 0
        for i in range(len(matrix)):
            for j in range(len(matrix[0])):
                if matrix[i][j] == "1":
                    if i == 0 or j == 0:
                        dp[i][j] = 1
                    else:
                        dp[i][j] = min(dp[i][j - 1], dp[i - 1][j], dp[i - 1][j - 1]) + 1
                    maxSide = max(maxSide, dp[i][j])
        return maxSide * maxSide

class Solution:
    def getIntersectionNode(self, headA: ListNode, headB: ListNode) -> ListNode:
        p1 = headA
        p2 = headB
        while p1 is not None or p2 is not None:
            if p1 == p2:
                return p1
            if p1 is None:
                p1 = headB
            else:
                p1 = p1.next
            if p2 is None:
                p2 = headA
            else:
                p2 = p2.next
        return None

class Solution:
    def mergeKLists(self, lists: List[Optional[ListNode]]) -> Optional[ListNode]:
        res = None
        for node in lists:
            res = self.merge(res, node)
        return res

    def merge(self, res, tmp):
        ret = ListNode()
        p = ret
        while res is not None and tmp is not None:
            if res.val < tmp.val:
                p.next = res
                res = res.next
            else:
                p.next = tmp
                tmp = tmp.next
            p = p.next
        p.next = res if tmp is None else tmp
        return ret.next

        if len(prices) < 2:
            return 0
        res = 0
        i = 0
        j = i + 1

                i = j

class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        res = []
        tmp = []
    
        def dfs(idx, sm):
            if idx >= len(candidates) or sm >= target:
                if sm == target:
                    res.append(tmp[:])
                return
            tmp.append(candidates[idx])
            dfs(idx, sm + candidates[idx])
            tmp.pop()
            dfs(idx + 1, sm)
        dfs(0, 0)
        return res

class Solution:
    def exist(self, board: List[List[str]], word: str) -> bool:
        self.res = False
        for i in range(len(board)):
            for j in range(len(board[0])):
                self.dfs(board, word, 0, i, j)
        return self.res
 
    def dfs(self, board, word, idx, i, j):
        if idx >= len(word):
            self.res = True
            return
        if i < 0 or j < 0 or i >= len(board) or j >= len(board[0]):
            return
        if board[i][j] != word[idx]:
            return
        ch = board[i][j]
        board[i][j] = '0'
        self.dfs(board, word, idx + 1, i + 1, j)
        self.dfs(board, word, idx + 1, i - 1, j)
        self.dfs(board, word, idx + 1, i, j + 1)
        self.dfs(board, word, idx + 1, i, j - 1)
        board[i][j] = ch

class Solution:
    def reverseBetween(self, head: ListNode, left: int, right: int) -> ListNode:
        res = ListNode(None, head)
        p1 = res
        p2 = head
        for _ in range(left - 1):
            p1 = p1.next
            p2 = p2.next
        for _ in range(right - left):
            tmp = p2.next
            p2.next = tmp.next if tmp else None
            tmp.next = p1.next
            p1.next = tmp
        return res.next

        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if self.grid[i][j] == '1':
                    self.dfs(i, j)
                    num += 1
        return num

    def dfs(self, i, j):
        if i < 0 or j < 0 or i >= len(self.grid) or j >= len(self.grid[0]):
            return
        if self.grid[i][j] != '1':
            return
        self.grid[i][j] = '0'
        self.dfs(i + 1, j)
        self.dfs(i - 1, j)
        self.dfs(i, j + 1)
        self.dfs(i, j - 1)
